#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    vector<int>nums(n),pos(n+1);
    for(int i=0;i<n;i++){
        cin>>nums[i];
        pos[nums[i]]=i;
    }
    int round=1;
    for(int x=2;x<=n;x++){
        if(pos[x]<pos[x-1]) round+=1;
    }
    cout<<round;
    return 0;

}
