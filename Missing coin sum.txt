#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cin>>n;
    long long x=1;
    vector<int>coins(n);
    for(int i=0;i<n;i++)cin>>coins[i];
    sort(coins.begin(),coins.end());
    for(int i=0;i<n;i++){
        if(coins[i]<=x) x+=coins[i];
        else break;
    }
    cout<<x;
    return 0;

}
