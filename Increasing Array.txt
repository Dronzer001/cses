#include<bits/stdc++.h>
using namespace std;
int main(){
    vector<int>arr;
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        int e;
        cin>>e;
        arr.push_back(e);
    }
    long long count=0;
    for(int i=1;i<n;i++){
        if(arr[i]<arr[i-1]){
            long long now=arr[i-1]-arr[i];
            arr[i]+=now;
            count+=now;
        }
    }
    cout<<count;
        return 0;
}
