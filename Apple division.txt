#include<bits/stdc++.h>
using namespace std;
long long solve(vector<int>&arr,int index,long long sum1,long long sum2){
    if(index==arr.size()){
        return abs(sum1-sum2);
    }
    long long s1=solve(arr,index+1,sum1+arr[index],sum2);
    long long s2=solve(arr,index+1,sum1,sum2+arr[index]);
    return min(s1,s2);
}
int main(){
    int n;
    cin>>n;
    vector<int>arr(n);
    for(int i=0;i<n;i++){
        cin>>arr[i];
    }
   cout<<solve(arr,0,0,0);
   return 0;
}
