#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,m,k;
    cin>>n>>m>>k;
    int count=0;
    vector<int>desired(n),actual(m);
    for(int i=0;i<n;i++) cin>>desired[i];
    for(int j=0;j<m;j++) cin>>actual[j];
    sort(actual.begin(),actual.end());
    sort(desired.begin(),desired.end());
    int i=0,j=0;
    while(i<n && j<m){
        int mini=desired[i]-k;
        int maxi=desired[i]+k;
        if(actual[j]<mini) j++;
        else if(actual[j]>maxi) i++;
        else{
            count+=1;
            i+=1;
            j+=1;
        }
    }
    cout<<count;
    return 0;
}
