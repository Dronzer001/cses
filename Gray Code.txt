#include<bits/stdc++.h>
using namespace std;
string binary(long long i,int n){
    string s;
    for(int j=n-1;j>=0;j--){
        s+=((i>>j)&1)?'1':'0';
    }
    return s;
}
int main(){
    int n;
    cin>>n;
    long long total=1LL<<n;//2^n
    for(long long i=0;i<total;i++){
        long long gray=i^(i>>1);//>> 1 means right shift by 1 bit & ^ is XOR (exclusive OR).
        cout<<binary(gray,n)<<"\n";
    }
    return 0;
}
