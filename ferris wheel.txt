#include<bits/stdc++.h>
using namespace std;
int main(){
    int n,x;
    cin>>n>>x;
    vector<int>weight(n);
    vector<bool>counted(n,false);
    for(int i=0;i<n;i++) cin>>weight[i];
    sort(weight.begin(),weight.end());
    int start=0,end=n-1,count=0;
    while(start<end){
        long long sum=weight[end]+weight[start];
        if(sum<=x){
            counted[start]=true;
            counted[end]=true;
            start++;
            end--;
        }
        else{
            counted[end]=true;
            end--;
        }
        count+=1;
    }
    if(!counted[start]) count+=1;
    cout<<count;
    return 0;
}
