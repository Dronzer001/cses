#include<bits/stdc++.h>
using namespace std;
int main(){
    string s;
    cin>>s;
    int count=1,curr=0;
    for(int i=1;i<s.length();i++){
        if(s[i]==s[i-1]){
            curr=1;
            while(s[i]==s[i-1]){
                curr+=1;
                i++;
            }
        }
        else{
            curr=1;
        }
        count=max(count,curr);
    }
    cout<<count<<endl;
    return 0;
}
